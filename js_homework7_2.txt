var services = {
  "стрижка": "60 грн",
  "гоління": "80 грн",
  "Миття голови": "100 грн"
};

// Додаємо нову послугу
services['Розбити скло'] = "200 грн";

// Виділення числа із рядка з валютою
function getPrice(service) {
  return parseFloat(services[service]); // Витягуємо число, ігноруючи валюту
}

// Загальна вартість всіх послуг
function price() {
  let total = 0;
  for (let service in services) {
      total += getPrice(service);
  }
  return total + " грн"; // Повертаємо значення з валютою
}

// Знаходження мінімальної вартості
function minPrice() {
  let prices = Object.values(services).map(price => parseFloat(price));
  return Math.min(...prices) + " грн"; // Повертаємо мінімальну ціну з валютою
}

// Знаходження максимальної вартості
function maxPrice() {
  let prices = Object.values(services).map(price => parseFloat(price));
  return Math.max(...prices) + " грн"; // Повертаємо максимальну ціну з валютою
}

// Перевірка
console.log("Загальна вартість послуг: " + price());
console.log("Мінімальна вартість послуги: " + minPrice());
console.log("Максимальна вартість послуги: " + maxPrice());
