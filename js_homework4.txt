let max = 1000;
let min = 100;
let numA = Array();

// Функція для генерації випадкових чисел
function generateRandomNumbers(min, max, count) {
    let numbers = [];
    for (let i = 0; i < count; i++) {
        let getRandom = Math.floor(Math.random() * (max - min + 1) + min);
        numbers.push(getRandom);
    }
    return numbers;
}

// Функція для підрахунку парних і непарних чисел та їх процентного співвідношення
function calculateEvenOddPercentage(numbers) {
    let evenCount = 0;
    let oddCount = 0;

    for (let i = 0; i < numbers.length; i++) {
        if (numbers[i] % 2 === 0) {
            evenCount++;
        } else {
            oddCount++;
        }
    }

    let evenPercentage = (evenCount / numbers.length) * 100;
    let oddPercentage = (oddCount / numbers.length) * 100;

    console.log("Процентне співвідношення парних чисел: " + evenPercentage.toFixed(2) + "%");
    console.log("Процентне співвідношення непарних чисел: " + oddPercentage.toFixed(2) + "%");
}

// Генерація масиву випадкових чисел
numA = generateRandomNumbers(min, max, 13);

console.log("Сгенеровані числа: ", numA);

// Підрахунок процентного співвідношення парних та непарних чисел
calculateEvenOddPercentage(numA);
