    }
  ]
};

function findValueByKey(companyObj, companyName) {
  // Перевіряємо, чи збігається назва компанії з шуканою
  if (companyObj.name === companyName) {
    return companyObj;
  }

  // Якщо є клієнти (subCompany), перебираємо їх
  if (companyObj.clients) {
    for (let client of companyObj.clients) {
      const result = findValueByKey(client, companyName); // Викликаємо рекурсію
      if (result) {
        return result;
      }
    }
  }

  // Якщо є партнери (subSubCompany), перебираємо їх
  if (companyObj.partners) {
    for (let partner of companyObj.partners) {
      const result = findValueByKey(partner, companyName); // Викликаємо рекурсію
      if (result) {
        return result;
      }
    }
  }

  // Якщо нічого не знайшли, повертаємо null
  return null;
}
